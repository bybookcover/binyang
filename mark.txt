
//添加远程版本库：
git remote add [shortname] [url]
shortname 为本地的版本库，例如：

//取消代理 
PS D:\vue1.0\binyang> git config --global --unset  https.proxy

//拉取远程仓库信息 git pull <远程地址> <远程分支名称>:<本地分支名称>
PS D:\vue1.0\binyang> git  pull https://github.com/bybookcover/binyang.git master:master

//从本地的分支版本上传到远程并合并 git push <远程地址> <远程分支名称>:<本地分支名称>
PS D:\vue1.0\binyang> git  push https://github.com/bybookcover/binyang.git master:master

//设置为默认push对象
To push the current branch and set the remote as upstream, use

    git push --set-upstream origin master
